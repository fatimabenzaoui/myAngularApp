authentification.service.ts

import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';

const link = 'http://localhost:3000/api/Users/login';

@Injectable({
  providedIn: 'root',
})
export class AuthentificationService {
  constructor(
    private http: HttpClient
  ) {}
  login(credentials) {
    return this.http.post(link, credentials);
  }
  logout() {
    localStorage.removeItem('token');
  }
  isLogged() {
    return !! localStorage.getItem('token');
  }
}

login.component.ts

import { Router } from '@angular/router';
import { AuthentificationService } from 
'./../authentification.service';
import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.css'],
})
export class LoginComponent implements OnInit {
  message = '';
  constructor(
    private authentificationService: AuthentificationService,
    private router: Router
  ){}
  ngOnInit(): void {}
  login(credentials) {
      this.authentificationService.login(credentials).subscribe(
        (response: any) => {
          const token = response.id;
          const link = ['contactapp'];
          localStorage.setItem('token', token);
          this.router.navigate(link);
        },
        (error) => {
          console.log('Erreur' + error);
        }
      );
  }
}
